#include <bits/stdc++.h>
using namespace std;
#define N 305
#define INF 0x7ffffff
class MaxFlow
{
private:
    int s;//起点
    int t;//终点
    int n;//整张图的节点数量
    int flow[N];//流到N号点的流量
    int pre[N];//N号点的前驱，用于记录增广路径并回溯更新剩余网络
    int G[N][N];//流网络的邻接矩阵
    int bfs();//用于寻找增广路径 如果找到，就返回增广的流量increase，如果找不到就返回-1
    void init();
public:
    int maxflow();//函数入口
    
};

函数

int MaxFlow::maxflow()
{
    int maxflow = 0;//用于记录整张图的最大流
    int increase=0;
    //初始化流网络
    this->init();
  //  cout << this->G[1][1] << endl;
    while ((increase = bfs())!=-1)
    {
        //更新剩余网络
        int now = t;
        while (now != s)
        {
            this->G[pre[now]][now] -= increase;
            this->G[now][pre[now]] += increase;
            now = pre[now];
        }
        //最大流增加增广的流量 为啥是加法捏
        maxflow += increase;
    }
    return maxflow;
}
void MaxFlow::init()
{
    int w;
    cin >> this->n >> this->s >> this->t >> w;
    for (int i = 0; i < this->n; i++)
        for (int j = 0; j < this->n; j++)
            this->G[i][j] = 0;
    for (int i = 0; i < w; i++)
    {
        int x, y;
        cin >> x >> y;
        cin >> this->G[x][y];
    }
    return;
}
int MaxFlow::bfs()
{
    //初始化
    for (int i = 0; i < N; i++)
    {
        pre[i] = -1;
    }
    for (int i = 0; i < N; i++)
    {
        flow[i] = 0;
    }
   
    //初始化s
    pre[s] = 0;
    flow[s] = INF;
    queue <int> Q;
    Q.push(s);
    while (!Q.empty())
    {
        int now;
        now = Q.front();
        Q.pop();
        if (now == this->t) break;
        for (int i = 0; i < this->n ; i++)
        {
            if (this->G[now][i] != 0 && pre[i] == -1)//双条件 还得是没访问过的
            {
                Q.push(i);
                flow[i] = min(flow[now], this->G[now][i]);//前面的管子细还是后面的管子细
                pre[i] = now;
            }       
        }
    }
    if (pre[t]==-1)
        return -1;//找不到返回-1
    else
        return this->flow[t];
}

#include<cstdio>
#include<cstring>
#include<algorithm>
#include<queue>
using namespace std;
const int INF=0x7ffffff;

queue <int> q;
int n,m,x,y,s,t,g[201][201],pre[201],flow[201],maxflow; 
//g邻接矩阵存图，pre增广路径中每个点的前驱，flow源点到这个点的流量 

inline int bfs(int s,int t)//虽然不是很清楚为什么是BFS不是DFS
{
    while (!q.empty()) q.pop();
    for (int i=1; i<=n; i++) pre[i]=-1;//每一轮都需要重置前驱，重置路径
    pre[s]=0;//初始化起点
    q.push(s);
    flow[s]=INF;//起点有无限的流量
    while (!q.empty())
    {
        int x=q.front();
        q.pop();
        if (x==t) break;//找到了捏
        for (int i=1; i<=n; i++)
          //EK一次只找一个增广路 
          if (g[x][i]>0 && pre[i]==-1)
          {
            pre[i]=x;
            flow[i]=min(flow[x],g[x][i]);//前面的管子细还是后面的管子细
            q.push(i);
          }
    }
    if (pre[t]==-1) return -1;//找不到返回-1
    else return flow[t];//能找到返回最大流
}

//increase为增广的流量 
void EK(int s,int t)
{
    int increase=0;
    while ((increase=bfs(s,t))!=-1)
    {//迭代 
        int k=t;
        while (k!=s)
        {
            int last=pre[k];//回溯更新剩余网络
            g[last][k]-=increase;
            g[k][last]+=increase;
            k=last;
        }
        maxflow+=increase;//最大流增加增广的流量 为啥是加法捏
    }
}

数据结构

#define N 300
#define INF 0x7ffffff
struct node
{
  int cap;
  int cost;
};
class MaxFlow
{
private:
  node G[N][N];
  int cost = 0;
  int n;
  int pre[N];
  int s;
  int t;
  void init();
  int spfa();
public:
  int maxflow();
  int mincost()
  {
    return this->cost;
  }
};

函数

void MaxFlow::init()
{
  for (int i = 0; i < N; i++)
  {
    this->pre[i] = -1;
    for (int j = 0; j < N; j++)
    {
      this->G[i][j].cap = 0;
      this->G[i][j].cost = INF;
    }
  }
  this->cost = 0;

  int m;//边的数量
  cin >> this->n >> this->s >> this->t >> m;
  //输入点的数量 起点 终点 和边的数量
  for (int i = 0; i < m; i++)
  {
    int x; int y;
    //输入每一条边
    cin >> x >> y;
    //输入每一条边的容量和费用
    cin >> this->G[x][y].cap >> this->G[x][y].cost;
    //在所有位置上都建立默认的反向边
    this->G[y][x].cost = -1 * this->G[x][y].cost;
  }

  return;
}
int MaxFlow::maxflow()
{
  int ans = 0;
  int increase = 0;
  this->init();
  while ((increase = this->spfa()) != -1)
  {
    int now;
    now = t;
    while (now != s)
    {
      G[pre[now]][now].cap -= increase;
      G[now][pre[now]].cap += increase;
      now = pre[now];
    }
    ans += increase;
  }
  return ans;
}
int MaxFlow::spfa()
{
  priority_queue <int> Q;//优先队列优化
  /*初始化*/
  int visit[N], dist[N], flow[N];
  for (int i = 0; i < this->n; i++)
  {
    visit[i] = 0;
    dist[i] = INF;
    flow[i] = INF;
  }
  /*塞入s*/
  dist[this->s] = 0;
  visit[this->s] = 1;
  Q.push(this->s);
  /*队列松弛*/
  while (!Q.empty())//在cap图存在边的情况下，对应选择费用图上的边作为新图 寻找s到t的单源最短路径（完全基于费用图）
  {
    int now;
    now = Q.top();
    Q.pop();
    visit[now] = 0;//回归0使其可以再次入队
    for (int i = 0; i < this->n; i++)
    {
      if (this->G[now][i].cap == 0) continue;//如果CAP图上都不不存在边的话，直接跳过
      //做SPFA松弛 注意先判断一下防止溢出
      if (this->G[now][i].cost != INF && dist[i] > dist[now] + this->G[now][i].cost)
      {
        dist[i] = dist[now] + this->G[now][i].cost;
          /*做关于流的处理 其中flow[i]中保存的是能流到i这个位置的流 
              对于flow[i]和now-i的这个管子填满做min*/
        flow[i] = min(flow[now], this->G[now][i].cap);
        //记录一下被选择的边
        this->pre[i] = now;
        if (visit[i] == 0)
        {
          visit[i] = 1;
          Q.push(i);
        }
      }        
    }
  }
  if (dist[this->t] == INF)
    return -1;
  else
  {
    this->cost += dist[this->t] * flow[this->t];
    return flow[this->t];
  }  
}
