void Arc::Init(int N)
{
    for (int i = 0; i < N; i++)
    {
        cin >> this->Edge[i].start >> this->Edge[i].end >> this->Edge[i].value;
    }
    this->Len = N;
    return;
}
void MGragh::Init(Arc A)//会初始化最短的边
{
    for (int i = 0; i < A.Len; i++)
    {
        if (arcs[A.Edge[i].start][A.Edge[i].end].adj != 0 && arcs[A.Edge[i].end][A.Edge[i].start].adj != 0 )
        {
            if (arcs[A.Edge[i].end][A.Edge[i].start].adj > A.Edge[i].value)
            {
                this->arcs[A.Edge[i].start][A.Edge[i].end].adj = A.Edge[i].value;
                this->arcs[A.Edge[i].end][A.Edge[i].start].adj = A.Edge[i].value;
            }
        }
        else
        {
            this->arcs[A.Edge[i].start][A.Edge[i].end].adj = A.Edge[i].value;
            this->arcs[A.Edge[i].end][A.Edge[i].start].adj = A.Edge[i].value;
        }
    }
}
