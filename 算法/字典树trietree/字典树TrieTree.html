类结构

class TrieTree
{
private:
    int id = 0;
    int trie[1000][26];
    int cnt[1000];
public:
    //插入新字符串
    void insert(string s);
    //寻找该字符串是否在树中
    bool find(string s);
    //寻找最小的字符串
    string findmin();
    //初始化
    void init();
};

函数内容

void TrieTree::insert(string s)
{
    int now = 0;
    for (int i = 0; i < s.length(); i++)
    {
        if (this->trie[now][s[i] - 'a'] == 0)
        {
            this->trie[now][s[i] - 'a'] = ++this->id;
        }
        now = this->trie[now][s[i] - 'a'];
    }
    cnt[now]++;
    return;
}
bool TrieTree::find(string s)
{
    int now = 0;
    for (int i = 0; i < s.length(); i++)
    {
        if (this->trie[now][s[i] - 'a'] == 0)
        {
            return false;
        }
        now = this->trie[now][s[i] - 'a'];
    }
    if (cnt[now] == 0) return false;
    return true;
}
string TrieTree::findmin()
{
    string ans;
    int now = 0;
    bool flag = 1;
    while (flag)
    {
        if (cnt[now] != 0) break;
        flag = 0;
        for (int i = 0; i <= 25; i++)
        {
            if (this->trie[now][i] != 0)
            {
                ans += char('a' + i);
                now = this->trie[now][i];
                flag = 1;
                break;
            }
        }
    }
    return ans;
}
void TrieTree::init()
{
    memset(this->trie, 0, sizeof(this->trie));
    memset(this->cnt, 0, sizeof(this->cnt));
}
