树状数组可以解决什么问题呢？

可以解决大部分区间上面的修改以及查询的问题，例如1.单点修改，单点查询，2.区间修改，单点查询，3.区间查询，区间修改，换言之，线段树能解决的问题，树状数组大部分也可以，但是并不一定都能解决，因为线段树的扩展性比树状数组要强.

初始化

void BiIndexTree::Init(int a[], int Len)
{
    this->N = Len;
    C = new int[N + 5];
    //初始化
    for (int i = 0; i < N + 5; i++)
    {
        C[i] = 0;
    }
    //一个一个更新到树状数组中
    for (int i = 0; i < Len; i++)
    {//a是从0开始存的，但是我们的c需要从1开始存
        this->Add(i + 1, a[i]);
    }
    return;
}

单点修改

void BiIndexTree::Add(int i, int value)
{
    while (i <= this->N)
    {//只需要改比自己更大的
        C[i] += value;
        i = i + Lowbit(i);
    }
    return;
}

前缀和查询（区间查询）

int BiIndexTree::Sum(int i)
{
    int sum = 0;
    for (i; i > 0; i -= this->Lowbit(i))
    {
        sum += this->C[i];
    }
    return sum;
}

int sum(int x, int y) {
  int res = 0;
  for (int i = x; i > 0; i -= lowbit(i)) {
      for (int j = y; j > 0; j -= lowbit(j)) {
          res += c[i][j];
      }
  }
  return res;
}

区间修改，区间查询

我们看到，由于我们目标记录的是数组a[1:n]，而实际存储的是d[1:n]，
那么已经实现了区间修改，如何完成区间查询呢？显然，区间查询的基础是快速求数组a[1:n]的前缀和，
显然数组a[1:n]的前缀和：
a[1]+a[2]+⋯+a[i]=d[1]×i+d[2]×(i−1)+⋯+d[i]×1a[1]+a[2]+\cdots+a[i]=d[1] \times i+d[2] \times(i-1)+\cdots+d[i] \times 1a[1]+a[2]+⋯+a[i]=d[1]×i+d[2]×(i−1)+⋯+d[i]×1
不难发现右侧可以化成：
d[1]×i+d[2]×(i−1)+⋯+d[i]×1=[d[1]×(i+1)+d[2]×(i+1)+⋯+d[i]×(i+1)]−[d[1]×1+d[2]×2+⋯+d[i]×i]=(i+1)×(d[1]+d[2]+⋯+d[i])−(d[1]×1+d[2]×2+⋯+d[i]×i)\begin{array}{l}d[1] \times i+d[2] \times(i-1)+\cdots+d[i] \times 1 \\ =[d[1] \times(i+1)+d[2] \times(i+1)+\cdots+d[i] \times(i+1)]-[d[1] \times 1+d[2] \times 2+\cdots+d[i] \times i] \\ =(i+1) \times(d[1]+d[2]+\cdots+d[i])-(d[1] \times 1+d[2] \times 2+\cdots+d[i] \times i)\end{array}d[1]×i+d[2]×(i−1)+⋯+d[i]×1=[d[1]×(i+1)+d[2]×(i+1)+⋯+d[i]×(i+1)]−[d[1]×1+d[2]×2+⋯+d[i]×i]=(i+1)×(d[1]+d[2]+⋯+d[i])−(d[1]×1+d[2]×2+⋯+d[i]×i)​
这样一来，我们就可以想到，在原来的数组C[i]记录d[i]的基础上，
再搞一个数组C2[i]记录d[i]×i即可。（当然，实际写代码的时候要明确，C数组和C2数组都是树状数组，不是原数组）

using namespace std;
class BiIndexTree
{
public:
    //求a[i]的前缀和
    long long int Sum(int i);
    //单点修改
    void Add(int i, int value);//i为修改的位置 从1开始存
  //初始化
  void init(long long int* a=NULL,int Len=0);//a[]为原来的数组，len为原来的数组长度
protected:
    long long int* C;//这是一个数组 记录树状数组
    int N;//记录长度
  
};
void BiIndexTree::init(long long int* a, int Len)
{
    this->N = Len;
    C = new long long int[N + 5];
    //初始化
    for (int i = 0; i < N + 5; i++)
    {
        C[i] = 0;
    }
    //一个一个更新到树状数组中
    for (int i = 0; i < Len; i++)
    {//a是从0开始存的，但是我们的c需要从1开始存
        this->Add(i + 1, a[i]);
    }
    return;
}
void BiIndexTree::Add(int i, int value)
{
    while (i <= this->N)
    {//只需要改比自己更大的
        C[i] += value;
        i = i + Lowbit(i);
    }
    return;
}
long long int BiIndexTree::Sum(int i)
{
    long long int sum = 0;
    for (i; i > 0; i -= Lowbit(i))
    {
        sum += this->C[i];
    }
    return sum;
}
class BiIndexTree2:public BiIndexTree
{
public:
  void init(long long int* a=NULL,int Len=0);
};
void BiIndexTree2::init(long long int* a,int Len)
{
    this->N = Len;
    C = new long long int[N + 5];
    for (int i = 0; i < N + 5; i++)
    {
        C[i] = 0;
    }
    for (int i = 0; i < Len; i++)
    {
        this->Add(i + 1, (i+1)*a[i]);
    }
    return;
}
class BIT
{
private:
  BiIndexTree C1;
  BiIndexTree2 C2;
  int N;
public:
  BIT(long long int* a=NULL,int len=0);
  ~BIT();
  void Add(int i,int value);
  long long int Sum(int i);
  long long int Range_Ask(int l,int r);
  void Update(int l,int r,int x);
};

BIT::BIT(long long int* a,int len)
{
  for(int i=len-1;i>=1;i--)
  {
    a[i]=a[i]-a[i-1];//要提前把a处理成差分数组
  }
  C1.BiIndexTree::init(a,len);
  C2.BiIndexTree2::init(a,len);
  return;
}

BIT::~BIT()
{
  return;
}
void BIT::Add(int i,int value)
{
  //单点修改
  C1.Add(i,value);
  C1.Add(i+1,-value);
  C2.Add(i,i*value);
  C2.Add(i+1,-(i+1)*value);
  return;
}

long long int BIT::Sum(int i)
{
  //前缀和查询
  return (i+1)*C1.Sum(i)-C2.Sum(i);
}

long long int BIT::Range_Ask(int l,int r)
{
  //区间查询
  return this->Sum(r)-this->Sum(l-1);
}
void BIT::Update(int l,int r,int x)
{
  //区间修改
  C1.Add(l,x);
  C1.Add(r+1,-x);
  C2.Add(l,l*x);
  C2.Add(r+1,-(r+1)*x);
  return;
}

#include <bits/stdc++.h>
#define maxn 50100  
#define INF 10010  
using namespace std;
int n,m;
int a[maxn];
long long tr1[maxn] , tr2[maxn];
  
inline int lowbit(int x) { return x & (-x); }
  
inline void Add(int pos , int val)
{
  long long val2 = 1ll * val * pos;
  while(pos <= n) { tr1[pos] += val; tr2[pos] += val2; pos += lowbit(pos); }
}
  
inline long long Ask1(int pos)
{
  long long Ans = 0;
  while(pos) { Ans += tr1[pos]; pos -= lowbit(pos); }
  return Ans;
}
  
inline long long Ask2(int pos)
{
  long long Ans = 0;
  while(pos) { Ans += tr2[pos]; pos -= lowbit(pos); }
  return Ans;
}
  
inline long long Ask(int l , int r)
{
  long long res1 = Ask1(r) * (r + 1) - Ask1(l-1) * l;
  long long res2 = Ask2(r) - Ask2(l-1);
  return res1 - res2;
}
