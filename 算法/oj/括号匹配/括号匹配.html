struct ans {
  int a;
  int b;
};

{
  ans t1, t2;

  stack<ans>s1;
  stack<int>s2;

  while (scanf("%s", in) != EOF)
  {
    for (int i = strlen(in) - 1; i >= 0; i--)
    {
      if (in[i] == ')')
        s2.push(i + 1);
      else if (!s2.empty())
      {
        t1.b = s2.top();
        t1.a = i + 1;
        s1.push(t1);
        s2.pop();
      }
    }

    int size = s1.size();

    for (int i = 0; i < size; i++)
    {
      t2 = s1.top();
      printf("%d %d\n", t2.a, t2.b);
      s1.pop();
    }
  }

}

#include <stdio.h>
#include <math.h>
#include <stdlib.h>
#include <string.h>
int s[100010];//要开全局变量，不然栈会溢出
int ans[100010];
int find[100010];
void stack_push(int a[],int * top,int n)
{
  a[(*top)++] = n;
  return;
}
int stack_empty(int a[], int* top)
{
  if ((*top) == 0)
  {
    return 1;
  }
  else
    return 0;
}
int stack_pop(int a[],int *top)
{
  if (!stack_empty(a, top))
  {
    (*top)--;
    return a[*top];
  }
  else
    return -1;
}
int stack_top(int a[], int* top)
{
  return a[*top];
}
void Brackets_Matching(char in[])
{
  int* top1;
  int* top2;
  top1 = (int*)malloc(sizeof(int));
  top2 = (int*)malloc(sizeof(int));
  (*top1) = 0;
  (*top2) = 0;
  int i = strlen(in) - 1;
  while (i >= 0)
  {
    if (in[i] == ')')
    {
      stack_push(s, top1, i + 1);
    }
    else
    {
      if (!stack_empty(s, top1))//这个处理比较关键 不然就会WA
      {
        int t;
        t = stack_pop(s, top1);
        stack_push(ans, top2, i + 1);
        find[i + 1] = t;
      }      
    }
    i--;
  }
  while (!stack_empty(ans, top2))
  {
    int t;
    t = stack_pop(ans, top2);
    printf("%d %d\n", t, find[t]);
  }
  //printf("%s", in);
  return;
}
