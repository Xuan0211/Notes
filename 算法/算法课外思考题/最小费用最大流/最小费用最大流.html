最小费用最大流

数据结构

#define N 300
#define INF 0x7ffffff
struct node
{
  int cap;
  int cost;
};
class MaxFlow
{
private:
  node G[N][N];
  int cost = 0;
  int n;
  int pre[N];
  int s;
  int t;
  void init();
  int spfa();
public:
  int maxflow();
  int mincost()
  {
    return this->cost;
  }
};

函数

void MaxFlow::init()
{
  for (int i = 0; i < N; i++)
  {
    this->pre[i] = -1;
    for (int j = 0; j < N; j++)
    {
      this->G[i][j].cap = 0;
      this->G[i][j].cost = INF;
    }
  }
  this->cost = 0;

  int m;//边的数量
  cin >> this->n >> this->s >> this->t >> m;
  //输入点的数量 起点 终点 和边的数量
  for (int i = 0; i < m; i++)
  {
    int x; int y;
    //输入每一条边
    cin >> x >> y;
    //输入每一条边的容量和费用
    cin >> this->G[x][y].cap >> this->G[x][y].cost;
    //在所有位置上都建立默认的反向边
    this->G[y][x].cost = -1 * this->G[x][y].cost;
  }

  return;
}
int MaxFlow::maxflow()
{
  int ans = 0;
  int increase = 0;
  this->init();
  while ((increase = this->spfa()) != -1)
  {
    int now;
    now = t;
    while (now != s)
    {
      G[pre[now]][now].cap -= increase;
      G[now][pre[now]].cap += increase;
      now = pre[now];
    }
    ans += increase;
  }
  return ans;
}
int MaxFlow::spfa()
{
  priority_queue <int> Q;//优先队列优化
  /*初始化*/
  int visit[N], dist[N], flow[N];
  for (int i = 0; i < this->n; i++)
  {
    visit[i] = 0;
    dist[i] = INF;
    flow[i] = INF;
  }
  /*塞入s*/
  dist[this->s] = 0;
  visit[this->s] = 1;
  Q.push(this->s);
  /*队列松弛*/
  while (!Q.empty())//在cap图存在边的情况下，对应选择费用图上的边作为新图 寻找s到t的单源最短路径（完全基于费用图）
  {
    int now;
    now = Q.top();
    Q.pop();
    visit[now] = 0;//回归0使其可以再次入队
    for (int i = 0; i < this->n; i++)
    {
      if (this->G[now][i].cap == 0) continue;//如果CAP图上都不不存在边的话，直接跳过
      //做SPFA松弛 注意先判断一下防止溢出
      if (this->G[now][i].cost != INF && dist[i] > dist[now] + this->G[now][i].cost)
      {
        dist[i] = dist[now] + this->G[now][i].cost;
          /*做关于流的处理 其中flow[i]中保存的是能流到i这个位置的流 
              对于flow[i]和now-i的这个管子填满做min*/
        flow[i] = min(flow[now], this->G[now][i].cap);
        //记录一下被选择的边
        this->pre[i] = now;
        if (visit[i] == 0)
        {
          visit[i] = 1;
          Q.push(i);
        }
      }        
    }
  }
  if (dist[this->t] == INF)
    return -1;
  else
  {
    this->cost += dist[this->t] * flow[this->t];
    return flow[this->t];
  }  
}
