6.部分背包问题

问题重述

有一个背包，背包容量是M=150。有7个物品，物品可以分割成任意大小。 要求尽可能让装入背包中的物品总价值最大，但不能超过总容量。
物品ABCDEFG
重量35306050401025
价值10403050354030

输入格式

第一行两个整数 N,TN,TN,T。
接下来 NNN 行，每行两个整数 mi,vim_i,v_imi​,vi​。

输出格式

一个实数表示答案，输出两位小数

算法思想：

贪心算法，优先选择性价比更高的物品放入背包。

代码

#include <bits/stdc++.h>
using namespace std;
struct node 
{
  float w;
  int m;
  int v;
}s[110];
bool cmp(node x,node y)
{
  if(x.w>y.w)
  {
    return true;  
  }
  else
    return false;  
}
int main()
{
  
  int N,T;
  cin>>N>>T;
  for(int i=1;i<=N;i++)
  {
    cin>>s[i].m>>s[i].v;
    s[i].w=(float)s[i].v/(float)s[i].m;//计算总的性价比 
  }
  sort(s+1,s+N+1,cmp);
  int now=1;
  float ans=0;
  while(T&&now<=N)
  {
    if(T>=s[now].m)
    {
      ans+=s[now].v;
      T-=s[now].m;
      now++;
    }
    else
    {
      ans+=s[now].w*T;
      break;
    }
  }
  printf("%.2f",ans);
  return 0;
}

7.照亮的山景

问题描述

在一片山的上空，高度为T处有N个处于不同位置的灯泡，如图。如果山的边界上某一点于某灯i的连线不经过山的其它点，我们称灯i可以照亮该点。开尽量少的灯，使得整个山景都被照亮。山被表示成有m个转折点的折线。
提示：照亮整个山景相当于照亮每一个转折点。

输入

输入可能包含多笔测试资料。
第1行:一个整数M (1<=M<=100,000),代表山棱折线的转折点个数。
第2行：N T
其中N代表灯泡个数,T代表所有灯泡的高度,该高度确保高于整座山最高转折点的高度。(1<=N, T<=100,000)
第3~M+2行:Xi Hi,依序代表转折点的水平座标以及垂直海拔高度。(-100,000<=Xi, Hi<=100,000) 所有转折点座标将以X001座0标由小到大排列。 第M+3行:有N个整数 B1 B2 ...BN,代表N个灯泡的水平座标,且依座标值由小到大排列。

输出

如果可以点亮某几盏灯而照亮所有转折点,请输出最少需要点亮的灯泡数量。
否则请输出 "you need more bulbs!"(不包含双引号)。

算法思想

如何照亮一个转折点？

延长该转折点的两条边交于高度T，得到一个区间（也就是图中的蓝色部分），那么在这个区间内的灯都可以照亮这个转折点。

根据转折点得到一系列区间

选择位于更多重叠区间的灯

删去灯及其被照亮的转折点

重复找灯、删转折点

当所有的转折点都被删去（照亮）后，算法结束

算法优化：

我们可以用“山”的思想，优化掉一些不需要被讨论的转折点区间

一座山需要被照亮，有以下两种可能

延长山的两侧线条，与灯所在的高度线交于某两点，将两点间的区间标记为蓝色（如图）

case1 在蓝色区域内有一盏灯

case2 在蓝色区域外左右两侧各有一盏灯

讨论“山头”转折点

如何判断一座山的山头？

选取波谷和山坡上的点依次相连计算斜率，然后取斜率最大的那条作为边界的边

用“山头”转折点讨论问题

我们用每座山的两侧将高度为T的数轴划分为不同区间
每次挑选位于更多区间的灯（以便于照亮更多的山头）
删去已选择的灯和被照亮的山
继续选择位于更多区间的灯
直到所有的山头都被照亮了，算法结束

#include <bits/stdc++.h>
using namespace std;
int M, N, T;//M，给出的坐标点的数量 N，灯泡的数量 T灯泡的高度
float a[100010][2];//区间
float b[100010];//灯泡坐标
int c[100010];//记录灯泡位于多少个重叠的区间内
int main()
{ 
  memset(c, 0, sizeof(c));
  cin >> M;
  float x1, x2, x3, y1, y2, y3;
  float X1, X2;
  if (M <= 2)
  {
    cin >> N >> T;
    //随便搞 反正一盏灯肯定够用 
    //假装输入一下
    for (int i = 1; i <= M; i++)
    {
      cin >> x1 >> y1;
    }
    for (int i = 1; i <= N; i++)
    {
      cin >> b[i];
    }
    if (N >= 1)
      cout << 1 << endl;
    else
    {
      cout << "you need more bulbs!" << endl;
    }
    return 0;
  }
  //否则M>2
  cin >> N >> T;
  cin >> x1 >> y1;
  cin >> x2 >> y2;
  for (int i = 3; i <= M; i++)
  {
    cin >> x3 >> y3;
    float k1, k2;
    k1 = (y2 - y1) / (x2 - x1);
    k2 = (y2 - y3) / (x2 - x3);
    X1 = x2 + k1 * (T - y2);
    X2 = x2 + k2 * (T - y2);
    if (X2 < X1)
      swap(X1, X2);
    a[i - 2][0] = X1;
    a[i - 2][1] = X2;
    x1 = x2;
    x2 = x3;
    y1 = y2;
    y2 = y3;
  }
  for (int i = 1; i <= N; i++)
  {
    cin >> b[i];
    for (int j = 1; j <= M - 2; j++)
    {
      if (a[j][0] <= b[i] && a[j][1] >= b[i])
      {
        c[i]++;
      }
    }
  }
  
  int ans = 0;
  while (1)
  {
    int maxx = 0;
    int k = 0;
    for (int i = 1; i <= N; i++)
    {
      if (maxx < c[i])
      {
        maxx = c[i];
        k = i;
      }
    }
    if (maxx == 0)
    {
      break;//出口
    }
    ans++;
    //删点删灯删区间
    for (int i = 1; i <= M - 2; i++)
    {
      if (a[i][0] <= b[k] && a[i][1] >= b[k])
      {
        a[i][1] = a[i][0] - 10000.0;
        //奇怪的排除区间方式
      }
    }
    memset(c, 0, sizeof(c));
    for (int i = 1; i <= N; i++)
    {
      for (int j = 1; j <= M - 2; j++)
      {
        if (a[j][0] <= b[i] && a[j][1] >= b[i])
        {
          c[i]++;
        }
      }
    }
  }
  int left = 0;
  for (int i = 1; i <= M - 1; i++)
  {
    if (a[i][0] < a[i][1])
    {
      left++;
    }
  }
  if (left > 0)
  {
    cout << "you need more bulbs!" << endl;
  }
  else
  {
    cout << ans << endl;
  }
  return 0;
}

问题描述

某教学大楼一层有n个教室，从左到右依次编号为1、2、…、n。现在要把一些课桌从某些教室搬到另外一些教室，每张桌子都是从编号较小的教室搬到编号较大的教室，每一趟，都是从左到右走，搬完一张课桌后，可以继续从当前位置或往右走搬另一张桌子。

输入数据：

先输入n、m，然后紧接着m行输入这m张要搬课桌的起始教室和目标教室。

输出数据：

最少需要跑几趟。

算法分析：

贪心算法，每次尽量在本房间再次出发继续往后搬。
如果没有，则找下一个房间的搬
如果还是没有 只能再跑一趟了。

代码：

加入了一些格式
#include <bits/stdc++.h>
#define INF 0x3f3f3f
using namespace std;
int in[1000][2];
int visit[1000];
int n,m;
int main()
{
  memset(visit,0,sizeof(visit));
  cout<<"输入所有的教室数量n"<<endl<<"n=";
  cin>>n; 
  cout<<"输入需要搬运的桌子数量m"<<endl<<"m=";
  cin>>m;
  for(int i=1;i<=m;i++)
  {
    cout<<"输入第"<<i<<"张桌子的起始教室:s=";
    cin>>in[i][0];
    cout<<"输入第"<<i<<"张桌子的目标教室:e=";
    cin>>in[i][1];
  } 
  cout<<"----------------------"<<endl;
  int ans=0;
  while(1)
  {
    int minn=INF;
    int k=0;
    //找起点 
    for(int i=1;i<=m;i++)
    {
      if(!visit[i])
      {
        if(in[i][0]<minn)
        {
          minn=in[i][0];
          k=i;
        }
      }
    }
    //出口，也就是所有桌子都被搬了 
    if(k==0) 
    {
      break;
    }
    visit[k]=1;
    cout<<in[k][0]<<"->"<<in[k][1];     
    while(1)
    {
      int n1=0;
      int n2=0;
      for(int i=1;i<=m;i++)
      {
        if(visit[i]==0)
        {
          if(in[i][0]==in[k][1])
          {
            n1=i;
            break;
          }
          if(in[i][0]==in[k][1]+1)
          {
            n2=i;
          }
        }
      
      }
      if(n1==0&&n2==0)
      {
        break;
      }
      if(n1!=0)
      {
        k=n1;
        visit[k]=1;
        cout<<" -- "<<in[k][0]<<"->"<<in[k][1]; 
      }
      else
      {
        k=n2;
        visit[k]=1;
        cout<<" -- "<<in[k][0]<<"->"<<in[k][1]; 
      }
    }
    ans++;
    cout<<endl;
  }
  cout<<"一共搬了"<<ans<<"趟"<<endl;
  return 0; 
}
