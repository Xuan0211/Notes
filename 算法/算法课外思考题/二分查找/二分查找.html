//在长度为len的a数组中寻找k 如果存在k则输出k的位置 如果不存在输出-1
int BinarySearch(int a[], int len, int k)
{
  int right, left, mid;
  sort(a, a + len);
  right = 0;
  left = len - 1;
  while (right <= left)
  {
    mid = (right + left) / 2;
    if (mid == k) return mid;
    if (mid < k)
      right = mid + 1;
    else
      left = mid - 1;
  }
  return -1;
}
