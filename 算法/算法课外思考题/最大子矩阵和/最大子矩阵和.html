  vector<int> maxSubArray(vector<int>& nums) 
  {
        int maxsum=INT_MIN;
        int dp = nums[0];
        vector<int> ans(2);//用来记录答案
        int start = 0;
        for(int i=1 ;i < nums.size(); i++)
        {
            if(dp > 0)
            {    //dp[i-1] > 0 时
                dp+=nums[i];
            }
            else
            {   //dp[i-1]<0时
                dp=nums[i];
                start = i;     //当nums[i]自立门户时候，我们记录下子序列的起始位置
            }
            if(dp > maxsum)
            { //更新答案
                maxsum = dp;
                ans[0] = start;//记录下起始和终止位置
                ans[1] = i;
            }  
        }
        return ans;
    }

vector<int> getMaxMatrix(vector<vector<int>>& matrix) 
{
    int n = matrix.size();
    int m = matrix[0].size();
    vector<int> ans(4);
    int maxsum = -1000000;
    for (int i = 0; i < n; i++)//枚举起始行
    {
        vector<int> b(m);
        for (int j = i; j < n; j++)//枚举结束行，也可以理解为在枚举矩阵的行数
        {
            int dp = 0;//此时dp已经被化简
            //接下来其实在枚举一个从第i行起始，(m-n+1)行的矩阵块
            //所以接下来只需要关注 起始列 和 结束列 在哪里就可以了
            int start = 0;//记录 起始列
            for (int k = 0; k < m; k++)//枚举 结束列
            {
                b[k] += matrix[j][k];//求每一列的前缀和
                if (dp > 0)
                {
                    dp += b[k];
                }
                else
                {
                    dp = b[k];
                    start = k;
                }
                if (dp > maxsum)
                {
                    maxsum = dp;
                    ans[0] = i;//左上角行号
                    ans[1] = start;//左上角列号
                    ans[2] = j;//右下角行号
                    ans[3] = k;//右下角列号
                }
            }
        }
    }
    return ans;
}
