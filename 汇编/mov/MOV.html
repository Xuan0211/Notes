立即数寻址

MOV AL, 5
MOV AX, 3064H
MOV AL, ‘A’

寄存器寻址

操作数在指定的寄存器中 
MOV AX, BX
MOV AL, BH
MOV AX, 3064H

直接寻址

MOV AX,[2000H]

寄存器简介寻址

MOV   AX, [BX]               
MOV   AX, ES:[BX]         
MOV   AX, [BP]
EA 在基址寄存器(BX/BP)或变址寄存器(SI/DI)中
不可以是AX、CX、DX

寄存器相对寻址

mov ax,[200+bx]
mov ax,200[bx]
mov ax,[bx].200
可以拿ds:[bx]或者ds:[200]当数组名
另外一个当下标，在数组内被访问

基址变址寻址

(ax)=((ds)∗16+(bx)+(si))(ax)=((ds)*16+(bx)+(si))(ax)=((ds)∗16+(bx)+(si))
mov ax,[bx][si]

相对基址变址寻址

(ax)=((ds)∗16+(bx)+(si)+idata) (ax)=((ds)*16+(bx)+(si)+idata) (ax)=((ds)∗16+(bx)+(si)+idata)
mov ax,[bx+200+si]
mov ax,[200+bx+si]
mov ax,200[bx][si]
mov ax,[bx].200[si]
mov ax,[bx][si].200
