7.1 and和or指令

and

应用：将操作对象的相应位设为0

mov al,01100011B
and al,11111110B;将al的第0位设为0
and al,10111111B;将al的第6位设为0

or

应用：将操作对象的相应位设为1

mov al,01100011B
or al,00000001B;将al的第0位设为1
or al,01000000B;将al的第6位设为1

7.2 ASCLL码

a:61H 97D 0110 0001B

A:41H 65D 0100 0001B

应用：

大小写转换

先判断原来是大写还是小写 在做转换
add al,20H;大写转小写
sub al,20H;小写转大写

大小写转换（逃课版）

观察这个表可以发现
二进制十进制十六进制图形二进制十进制十六进制图形
0100 00016541A0110 00019761a
0100 00106642B0110 00109862b
0100 00116743C0110 00119963c
0100 01006844D0110 010010064d
0100 01016945E0110 010110165e
除了从右往左第0-4位 大小写的二进制完全一样
所以也可以不判断原来的字母是大写还是小写
如果要全部转大写：
and al,11011111B #或者0DFH
如果要全部转小写：
or al,001000000B

7.7 SI和DI

SI（Source Index）

源变址寄存器可用来存放相对于DS段之源变址指针；

DI（Destination Index）

目的变址寄存器，可用来存放相对于 ES 段之目的变址指针。

应用：复制字符串

让SI指向源字符串
让DI指向目的字符串
注意这里是用AX做中间寄存器进行数据传输的
所以一次要跳两格
如果用AL 或者AH那就是一次一格

实验6：实践课程中的程序7.9

datas segment
    db '1. display      '
    db '2. brows        '
    db '3. replace      '
    db '4. modify       '
datas ends

stacks segment
    dw 0,0,0,0,0,0,0,0;储存多重循环CX变量的栈段
stacks ends

codes segment
    assume cs:codes,ds:datas,ss:stacks
start:
    ;初始化ds
    mov ax,datas
    mov ds,ax
    
    ;初始化栈
    mov ax,stacks
    mov ss,ax
    mov sp,16
    
    mov bx,0
    mov cx,4;
  s:
  
    ;把cx塞进栈里
    push cx
    
    mov si,0  
    mov cx,4
  s0:
    mov al,[bx][si].3
    and al,110111111B
    mov [bx][si].3,al
    
    inc si
    loop s0
    
    ;恢复cx
    pop cx
    
    add bx,16
    loop s
       
    
    mov ah,4ch
    int 21h
codes ends
    end start
